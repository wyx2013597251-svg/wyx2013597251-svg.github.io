    let star=document.createElement("div");
    star.className="star";
    star.style.top=Math.random()*100+"vh";
    star.style.left=Math.random()*100+"vw";
    document.body.appendChild(star);
}

// 爱心
setInterval(()=>{
    let heart=document.createElement("div");
    heart.className="heart";
    heart.innerHTML="💗";
    heart.style.left=Math.random()*100+"vw";
    document.body.appendChild(heart);
    setTimeout(()=>heart.remove(),6000);
},300);

// 打字机
let message=`臭一二你什么时候才可以回来啊，布布真的真的好想好想你，不知道你过的怎么样，布布真的要崩溃了，没有食欲，什么吃的都不香，真的如果你再不回来我就要崩溃了，有时候晚上睡觉都想你什么时候回来，昨天听你说你今天下午回来我高兴的差点跳起来，今天你又说不知道，本布要崩溃了，差点就哭出来了，你回来好不好，我给你揉肚子，你想吃什么带我去吃，你不在的日子我真的好无聊，整天和那些人聊天我都要吐了，刚刚你生气不是不哄你，而是想给你这个惊喜，我怎么会不哄你，你又来月经了，更应该哄了，看到这些赶紧打电话给布布吧，他已经要寂寞死了。`;

let i=0;
function typing(){
    if(i<message.length){
        document.getElementById("text").innerHTML+=message.charAt(i);
        i++;
        setTimeout(typing,70);
    }else{
        document.getElementById("callBtn").style.display="block";
    }
}

// 延迟开始打字（等页面出来）
setTimeout(typing,6000);
